 # 更新日志
 
+## [4.0.0] - 2026-03-18
+
+### 变更
+
+- 💥 **默认交互入口切换为 WebUI（破坏性变更）**
+  - 默认启动路径从本地 OpenCV 窗口切换为 FastAPI WebUI，主交互改为浏览器 Canvas 渲染
+  - 检测控制、参数调整、历史消息与证据查看统一迁移到 WebUI 面板
+- ✨ **WebUI 证据导出与查看能力增强**
+  - 新增 WebUI 事件证据可选导出，支持按开关分别导出事件视频与原始帧序列
+  - 新增 `outputs/events/raw_frames/<event_id>/` 原始帧证据目录与 `frames.json` 时间戳清单
+  - WebUI 新增最近导出证据列表，支持展开查看清单并跳转截图、清单和单帧证据
+
+### 修复
+
+- 🐛 **修复 WebUI 事件链路与证据体验问题**
+  - 补回 WebUI 告警写回 `outputs/events` 的事件落盘链路，并增加最小化防抖/连续事件保持逻辑
+  - 修复 WebUI 事件收尾阶段的状态字段不完整导致的 `/api/webui/infer` 500 错误
+  - 修复 WebUI 证据视频导出速度偏快的问题，改为基于实际送帧时间估算导出帧率
+

+def _json_safe(value: Any) -> Any:
+    """把 numpy/枚举/复杂对象转换成可 JSON 化结构。"""
+    if value is None or isinstance(value, (str, int, float, bool)):
+        return value
+    if isinstance(value, np.generic):
+        return value.item()
+    if isinstance(value, np.ndarray):
+        return value.tolist()
+    if isinstance(value, (list, tuple)):
+        return [_json_safe(v) for v in value]
+    if isinstance(value, dict):
+        return {str(k): _json_safe(v) for k, v in value.items()}
+    if hasattr(value, "value"):  # Enum-like
+        return _json_safe(getattr(value, "value"))
+    if hasattr(value, "__dict__"):
+        return _json_safe(value.__dict__)
+    return str(value)

+def _extract_landmark_points(
+    pose_landmarks: Any,
+    bbox: Optional[List[int]],
+    frame_width: int,
+    frame_height: int,
+) -> List[Dict[str, Any]]:
+    """把 landmarks 统一转换为整帧像素坐标。"""
+    if pose_landmarks is None:
+        return []
+
+    raw = pose_landmarks
+    if isinstance(pose_landmarks, dict):
+        if isinstance(pose_landmarks.get("landmark"), list):
+            raw = pose_landmarks["landmark"]
+        elif isinstance(pose_landmarks.get("landmarks"), list):
+            raw = pose_landmarks["landmarks"]
+        elif isinstance(pose_landmarks.get("keypoints"), list):
+            raw = pose_landmarks["keypoints"]
+    else:
+        # 兼容 MediaPipe/自定义 LandmarkList 对象
+        if hasattr(pose_landmarks, "landmark") and isinstance(pose_landmarks.landmark, list):
+            raw = pose_landmarks.landmark
+        elif hasattr(pose_landmarks, "landmarks") and isinstance(pose_landmarks.landmarks, list):
+            raw = pose_landmarks.landmarks
+        elif hasattr(pose_landmarks, "keypoints") and isinstance(pose_landmarks.keypoints, list):
+            raw = pose_landmarks.keypoints
+
+    if not isinstance(raw, list):
+        return []
+
+    bbox_x1 = bbox[0] if bbox else 0
+    bbox_y1 = bbox[1] if bbox else 0
+    bbox_w = max(1, (bbox[2] - bbox[0])) if bbox else max(1, frame_width)
+    bbox_h = max(1, (bbox[3] - bbox[1])) if bbox else max(1, frame_height)
+
+    points: List[Dict[str, Any]] = []
+    for idx, point in enumerate(raw):
+        x = None
+        y = None
+        visibility = None
+        if isinstance(point, dict):
+            x = point.get("x")
+            y = point.get("y")
+            visibility = point.get("visibility")
+        elif hasattr(point, "x") and hasattr(point, "y"):
+            x = getattr(point, "x", None)
+            y = getattr(point, "y", None)
+            visibility = getattr(point, "visibility", None)
+        elif isinstance(point, (list, tuple)) and len(point) >= 2:
+            x = point[0]
+            y = point[1]
+            if len(point) >= 3:
+                visibility = point[2]
+
+        try:
+            px = float(x)
+            py = float(y)
+        except Exception:
+            continue
+
+        normalized = 0.0 <= px <= 1.2 and 0.0 <= py <= 1.2
+        if normalized:
+            frame_x = bbox_x1 + px * bbox_w
+            frame_y = bbox_y1 + py * bbox_h
+        else:
+            frame_x = px
+            frame_y = py
+
+        points.append(
+            {
+                "index": idx,
+                "x": float(frame_x),
+                "y": float(frame_y),
+                "visibility": float(visibility) if visibility is not None else None,
+            }
+        )
+    return points
+

+def _build_frame_ui_result(result: Dict[str, Any], frame: np.ndarray, frame_index: int) -> FrameUIResult:
+    """从检测结果构建前端渲染契约。"""
+    h, w = frame.shape[:2]
+    person_bboxes: List[List[int]] = []
+    raw_bboxes = result.get("all_person_bboxes") or []
+    if not raw_bboxes and result.get("person_bbox"):
+        raw_bboxes = [result.get("person_bbox")]
+    for bbox in raw_bboxes:
+        if not bbox:
+            continue
+        try:
+            x1, y1, x2, y2 = bbox
+            person_bboxes.append([int(x1), int(y1), int(x2), int(y2)])
+        except Exception:
+            continue
+
+    suspicious_regions: List[Dict[str, Any]] = []
+    for region in (result.get("suspicious_regions") or []):
+        try:
+            region_type, bbox, confidence = region
+            x1, y1, x2, y2 = bbox
+            suspicious_regions.append(
+                {
+                    "type": str(region_type),
+                    "bbox": [int(x1), int(y1), int(x2), int(y2)],
+                    "confidence": float(confidence),
+                }
+            )
+        except Exception:
+            continue
+
+    raw_pose_landmarks = result.get("all_pose_landmarks") or []
+    pose_people: List[Dict[str, Any]] = []
+    pose_landmarks: List[Any] = []
+    max_len = max(len(person_bboxes), len(raw_pose_landmarks))
+    for idx in range(max_len):
+        bbox = person_bboxes[idx] if idx < len(person_bboxes) else None
+        raw_pose = raw_pose_landmarks[idx] if idx < len(raw_pose_landmarks) else None
+        points = _extract_landmark_points(raw_pose, bbox, w, h)
+        pose_people.append(
+            {
+                "person_index": idx,
+                "bbox": bbox,
+                "coordinate_space": "frame_pixels",
+                "keypoints": points,
+            }
+        )
+        pose_landmarks.append(points)
+
+    fps = None
+    if detector and detector.performance_monitor:
+        fps = detector.performance_monitor.get_avg_fps()
+
+    return FrameUIResult(
+        timestamp=datetime.now().isoformat(),
+        frame_index=frame_index,
+        frame_size={"width": int(w), "height": int(h)},
+        cheating_detected=bool(result.get("cheating_detected", False)),
+        alerts=[str(x) for x in (result.get("alerts") or [])],
+        fusion_score=float(result.get("fusion_score", 0.0) or 0.0),
+        detection_time=float(result.get("detection_time", 0.0) or 0.0),
+        max_alert_level=_json_safe(result.get("max_alert_level")),
+        classified_alerts=_json_safe(result.get("classified_alerts") or []),
+        person_bboxes=person_bboxes,
+        suspicious_regions=suspicious_regions,
+        pose_landmarks=pose_landmarks,
+        pose_people=pose_people,
+        fsm_results=_json_safe(result.get("fsm_results") or {}),
+        performance={
+            "fps": float(fps) if fps is not None else None,
+            "queue_size": detector.result_queue.qsize() if detector else 0,
+        },
+    )
+

+def _apply_threshold(payload: FrameUIResult, threshold: float) -> FrameUIResult:
+    """按融合分数阈值更新作弊判定，便于 WebUI 调参。"""
+    if payload.fusion_score >= float(threshold):
+        return payload

+    data = payload.model_dump()
+    data["cheating_detected"] = False
+    data["alerts"] = []
+    data["classified_alerts"] = []
+    data["max_alert_level"] = None
+    return FrameUIResult(**data)

+def _extract_fsm_state(result: Dict[str, Any]) -> Optional[str]:
+    """提取一个可读的 FSM 状态摘要。"""
+    fsm_results = result.get("fsm_results") or {}
+    if not isinstance(fsm_results, dict):
+        return None
+    for _, fsm_result in fsm_results.items():
+        if not isinstance(fsm_result, dict):
+            continue
+        state_info = fsm_result.get("state_info") or {}
+        if isinstance(state_info, dict) and state_info.get("current_state"):
+            return str(state_info.get("current_state"))
+    return None

+def _append_webui_frame(frame: np.ndarray, current_time: float) -> None:
+    """缓存 WebUI 最近帧用于导出证据视频。"""
+    if frame is None:
+        return
+    webui_evidence_buffer.append((current_time, frame.copy()))
+    min_time = current_time - webui_evidence_buffer_seconds
+    while webui_evidence_buffer and webui_evidence_buffer[0][0] < min_time:
+        webui_evidence_buffer.popleft()
+
+
+def _capture_webui_pre_frames(trigger_time: float) -> List[Any]:
+    """截取事件触发前的缓存帧样本（时间戳 + 帧）。"""
+    start_time = trigger_time - webui_evidence_pre_seconds
+    return [(ts, frame.copy()) for ts, frame in webui_evidence_buffer if start_time <= ts <= trigger_time]
+
+
+def _estimate_webui_export_fps(samples: List[Any]) -> float:
+    """按 WebUI 实际送帧时间估算导出视频帧率，避免快放。"""
+    if not samples:
+        interval_ms = max(50.0, float(webui_settings.get("infer_interval_ms", 120) or 120))
+        return max(1.0, min(webui_runtime_fps, 1000.0 / interval_ms))
+
+    timestamps = [float(ts) for ts, _ in samples if ts is not None]
+    if len(timestamps) < 2:
+        interval_ms = max(50.0, float(webui_settings.get("infer_interval_ms", 120) or 120))
+        return max(1.0, min(webui_runtime_fps, 1000.0 / interval_ms))
+
+    duration = max(0.001, timestamps[-1] - timestamps[0])
+    fps = (len(timestamps) - 1) / duration
+    return max(1.0, min(webui_runtime_fps, float(fps)))
+

+        "max_alert_level": _json_safe(event_payload.get("max_alert_level")),
+        "fusion_score": float(event_payload.get("fusion_score") or 0.0),
+    })

+def _collect_recent_webui_exports(limit: int = 8) -> List[Dict[str, Any]]:
+    """读取最近导出的证据包列表。"""
+    if not file_manager:
+        return []
+    packages_dir = Path(file_manager.evidence_packages_dir)
+    if not packages_dir.exists():
+        return []
+
+    items: List[Dict[str, Any]] = []
+    package_files = sorted(packages_dir.glob("*.json"), key=lambda path: path.stat().st_mtime, reverse=True)
+    for path in package_files[: max(1, int(limit))]:
+        try:
+            data = json.loads(path.read_text(encoding="utf-8"))
+            alerts = data.get("alerts") or []
+            items.append({
+                "event_id": data.get("event_id") or data.get("evidence_id") or path.stem,
+                "timestamp": data.get("timestamp"),
+                "alerts": alerts,
+                "summary": " | ".join(str(item) for item in alerts[:3]),
+                "clip_path": data.get("clip_path"),
+                "raw_frames_dir": data.get("raw_frames_dir"),
+                "frame_manifest_path": data.get("frame_manifest_path"),
+                "screenshot_path": data.get("screenshot_path"),
+                "created_at": data.get("created_at"),
+            })
+        except Exception:
+            continue
+    return items

+def _finalize_webui_event(current_time: float) -> Dict[str, Any]:
+    """在后置窗口结束后导出 WebUI 证据视频。"""
+    global webui_active_event
+    if not file_manager or not webui_active_event:
+        return _empty_webui_event_state()
+
+    active = webui_active_event
+    samples = list(active.get("pre_frames", [])) + list(active.get("post_frames", []))
+    frames = [frame for _, frame in samples]
+    evidence_ref = None
+    if frames:
+        export_fps = _estimate_webui_export_fps(samples)
+        if logger:
+            logger.info(
+                f"WebUI事件导出: {active['event_id']} samples={len(samples)}, export_fps={export_fps:.2f}"
+            )
+        evidence_ref = file_manager.save_event_evidence_package(
+            event_id=active["event_id"],
+            timestamp=active["timestamp"],
+            frames=frames,
+            fps=export_fps,
+            event_payload=active["event_payload"],
+            source_mode="webui",
+            window_seconds={
+                "pre": webui_evidence_pre_seconds,
+                "post": max(0.0, current_time - float(active.get("trigger_time", current_time))),
+            },
+            frame_samples=samples,
+            export_clip=bool(webui_settings.get("export_event_video", True)),
+            export_raw_frames=bool(webui_settings.get("export_raw_frames", True)),
+        )
+        if evidence_ref:
+            file_manager.attach_evidence_to_event(active["timestamp"], evidence_ref)
+
+    result = _empty_webui_event_state()
+    result.update({
+        "event_finished": True,
+        "event_persisted": bool(evidence_ref),
+        "event_id": active["event_id"],
+        "event_clip_path": evidence_ref.get("clip_path") if evidence_ref else None,
+        "event_raw_frames_path": evidence_ref.get("raw_frames_dir") if evidence_ref else None,
+    })
+    webui_active_event = None
+    return result
+

+def _advance_webui_event_state(result: Dict[str, Any], payload: FrameUIResult, frame: np.ndarray, current_time: float) -> Dict[str, Any]:
+    """维护 WebUI 连续事件状态，避免重复刷盘并补齐证据视频。"""
+    global webui_active_event
+
+    state = _empty_webui_event_state()
+    if frame is None:
+        return state
+
+    _append_webui_frame(frame, current_time)
+
+    active = webui_active_event
+    if active is not None:
+        active["post_frames"].append((current_time, frame.copy()))
+
+    has_signal = bool(payload.cheating_detected and payload.alerts)
+    signature = " | ".join(sorted(str(alert) for alert in payload.alerts)) if has_signal else ""
+
+    if has_signal:
+        if active is None:
+            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S_%f")[:-3]
+            event_id = f"evt_{timestamp}"
+            saved_timestamp = file_manager.save_cheating_event(
+                timestamp=timestamp,
+                alerts=list(payload.alerts),
+                frame=frame,
+                cheating_detected=True,
+                confidence=float(payload.fusion_score or 0.0),
+                classified_alerts=result.get("classified_alerts") or [],
+                max_alert_level=result.get("max_alert_level"),
+                fusion_score=float(payload.fusion_score or 0.0),
+                fsm_state=_extract_fsm_state(result),
+                source_mode="webui",
+                event_id=event_id,
+            ) if file_manager else None
+            if saved_timestamp:
+                webui_active_event = {
+                    "event_id": event_id,
+                    "timestamp": timestamp,
+                    "signature": signature,
+                    "trigger_time": current_time,
+                    "hold_until": current_time + webui_alert_hold_seconds,
+                    "complete_after": current_time + webui_evidence_post_seconds,
+                    "pre_frames": _capture_webui_pre_frames(current_time),
+                    "post_frames": [],
+                    "event_payload": _build_webui_event_payload(result, payload, event_id, frame),
+                }
+                state["event_started"] = True
+                state["event_id"] = event_id
+            return state
+
+        active["complete_after"] = current_time + webui_evidence_post_seconds
+        active["hold_until"] = current_time + webui_alert_hold_seconds
+        if signature:
+            active["signature"] = signature
+        active["event_payload"]["alerts"] = list(payload.alerts)
+        active["event_payload"]["classified_alerts"] = result.get("classified_alerts") or []
+        active["event_payload"]["max_alert_level"] = result.get("max_alert_level")
+        active["event_payload"]["fusion_score"] = float(payload.fusion_score or 0.0)
+        active["event_payload"]["confidence"] = float(payload.fusion_score or 0.0)
+        active["event_payload"]["fsm_state"] = _extract_fsm_state(result)
+        return state
+
+    if active is not None and current_time < float(active.get("hold_until", current_time)):
+        return _build_webui_hold_state(active)
+
+    if active is not None and current_time >= float(active.get("complete_after", current_time)):
+        return _finalize_webui_event(current_time)
+    return state
+

+def _flush_webui_event(force: bool = False) -> Dict[str, Any]:
+    """在停止或关闭时强制导出未完成的 WebUI 事件。"""
+    if not webui_active_event:
+        return _empty_webui_event_state()
+    current_time = time.time()
+    if force:
+        return _finalize_webui_event(current_time)
+    if current_time >= float(webui_active_event.get("complete_after", current_time)):
+        return _finalize_webui_event(current_time)
+    return _empty_webui_event_state()

+function pushHistoryMessage(text, options = {}) {
+  const message = String(text || "").trim();
+  if (!message) {
+    return;
+  }
+  const cooldownMs = Number(options.cooldownMs || 0);
+  const now = Date.now();
+  const lastTime = Number(historyMessageTimes.get(message) || 0);
+  if (cooldownMs > 0 && now - lastTime < cooldownMs) {
+    return;
+  }
+  const last = historyMessages[0];
+  if (last && last.text === message) {
+    return;
+  }
+  historyMessages.unshift({
+    text: message,
+    time: new Date().toLocaleTimeString("zh-CN", { hour12: false }),
+  });
+  historyMessageTimes.set(message, now);
+  historyMessages = historyMessages.slice(0, 80);
+  renderHistory();
+}

+async function ensureExportManifest(item) {
+  const manifestPath = normalizeOutputPath(item?.frame_manifest_path);
+  if (!manifestPath || exportManifestCache.has(manifestPath)) {
+    return;
+  }
+  const manifestUrl = toOutputUrl(manifestPath);
+  if (!manifestUrl) {
+    return;
+  }
+  try {
+    const response = await fetch(manifestUrl);
+    if (!response.ok) {
+      throw new Error(await response.text());
+    }
+    const data = await response.json();
+    exportManifestCache.set(manifestPath, data);
+  } catch (err) {
+    console.error(err);
+    exportManifestCache.set(manifestPath, { error: "清单加载失败" });
+  }

+function drawPayload(payload) {
+  latestPayload = payload;
+  drawFrameSnapshot();
+  const ctx = canvasEl.getContext("2d");
+  const w = canvasEl.width;
+  const h = canvasEl.height;
+  ctx.clearRect(0, 0, w, h);
+
+  const srcW = payload.frame_size?.width || w;
+  const srcH = payload.frame_size?.height || h;
+  const transform = getContainTransform(w, h, srcW, srcH);
+
+  if (settings.draw_person_bbox) {
+    const bboxes = payload.person_bboxes || [];
+    ctx.strokeStyle = "#26d07c";
+    ctx.lineWidth = 2;
+    bboxes.forEach((bbox, idx) => {
+      const [x1, y1, x2, y2] = bbox;
+      const [cx1, cy1] = toCanvasPoint(x1, y1, srcW, srcH, transform, false);
+      const [cx2, cy2] = toCanvasPoint(x2, y2, srcW, srcH, transform, false);
+      const bw = cx2 - cx1;
+      const bh = cy2 - cy1;
+      if (bw <= 1 || bh <= 1) {
+        return;
+      }
+      ctx.strokeRect(cx1, cy1, bw, bh);
+      ctx.fillStyle = "rgba(38, 208, 124, 0.85)";
+      ctx.fillRect(cx1, cy1 - 20, 72, 18);
+      ctx.fillStyle = "#08150f";
+      ctx.font = "12px sans-serif";
+      ctx.fillText(`P${idx}`, cx1 + 5, cy1 - 7);
+    });
+  }
+
+  if (settings.draw_suspicious_regions) {
+    const regions = payload.suspicious_regions || [];
+    regions.forEach((item) => {
+      const [x1, y1, x2, y2] = item.bbox;
+      ctx.strokeStyle = "#ff8a65";
+      ctx.lineWidth = 2;
+      const [cx1, cy1] = toCanvasPoint(x1, y1, srcW, srcH, transform, false);
+      const [cx2, cy2] = toCanvasPoint(x2, y2, srcW, srcH, transform, false);
+      const bw = cx2 - cx1;
+      const bh = cy2 - cy1;
+      if (bw <= 1 || bh <= 1) {
+        return;
+      }
+      ctx.strokeRect(cx1, cy1, bw, bh);
+      ctx.fillStyle = "rgba(255, 138, 101, 0.92)";
+      const label = `${item.type} ${(item.confidence || 0).toFixed(2)}`;
+      ctx.fillRect(cx1, cy1 - 20, Math.max(112, label.length * 7), 18);
+      ctx.fillStyle = "#180f0d";
+      ctx.fillText(label, cx1 + 4, cy1 - 7);
+    });
+  }
+
+  if (settings.draw_skeleton) {
+    const poses = Array.isArray(payload.pose_people) && payload.pose_people.length
+      ? payload.pose_people
+      : buildFallbackPosePeople(payload);
+    ctx.fillStyle = "#38b8ff";
+    ctx.strokeStyle = "rgba(56, 184, 255, 0.85)";
+    ctx.lineWidth = 2;
+    poses.forEach((person) => {
+      const points = extractPosePoints(person);
+      if (!points.length) {
+        return;
+      }
+      const personLabel = resolvePersonLabel(person);
+      const coordinateSpace = resolveCoordinateSpace(person, points);
+      const poseEdges = resolvePoseEdges(points.length);
+      const visibleIndexSet = resolveVisibleIndexSet(points.length);
+
+      // 按标准骨架拓扑连线，避免“全点顺序乱连”
+      poseEdges.forEach(([a, b]) => {
+        const pa = points[a];
+        const pb = points[b];
+        if (!visibleIndexSet.has(a) || !visibleIndexSet.has(b)) {
+          return;
+        }
+        if (!isRenderablePoint(pa) || !isRenderablePoint(pb)) {
+          return;
+        }
+        const [ax, ay] = toCanvasPoint(pa.x, pa.y, srcW, srcH, transform, coordinateSpace);
+        const [bx, by] = toCanvasPoint(pb.x, pb.y, srcW, srcH, transform, coordinateSpace);
+        if (!Number.isFinite(ax) || !Number.isFinite(ay) || !Number.isFinite(bx) || !Number.isFinite(by)) {
+          return;
+        }
+        ctx.beginPath();
+        ctx.moveTo(ax, ay);
+        ctx.lineTo(bx, by);
+        ctx.stroke();
+      });
+
+      points.forEach((pt, index) => {
+        if (!visibleIndexSet.has(index)) {
+          return;
+        }
+        if (!isRenderablePoint(pt)) {
+          return;
+        }
+        const [cx, cy] = toCanvasPoint(pt.x, pt.y, srcW, srcH, transform, coordinateSpace);
+        if (!Number.isFinite(cx) || !Number.isFinite(cy)) {
+          return;
+        }
+        ctx.beginPath();
+        ctx.arc(cx, cy, 2.6, 0, 2 * Math.PI);
+        ctx.fill();
+
+        if (settings.draw_point_index) {
+          ctx.fillStyle = "#ffffff";
+          ctx.font = "11px monospace";
+          ctx.fillText(String(index), cx + 4, cy - 4);
+          ctx.fillStyle = "#38b8ff";
+        }
+      });
+
+      if (settings.draw_pose_label) {
+        const anchor = findPoseAnchor(points, visibleIndexSet);
+        if (anchor) {
+          const [lx, ly] = toCanvasPoint(anchor.x, anchor.y, srcW, srcH, transform, coordinateSpace);
+          ctx.fillStyle = "rgba(8, 21, 15, 0.92)";
+          ctx.fillRect(lx + 8, ly - 18, 34, 16);
+          ctx.fillStyle = "#7cf5c8";
+          ctx.font = "11px monospace";
+          ctx.fillText(personLabel, lx + 12, ly - 6);
+          ctx.fillStyle = "#38b8ff";
+        }
+      }
+    });
+  }
+
+  renderAlerts(payload.alerts || []);
+  const alertSignature = (payload.alerts || []).join(" | ");
+  if (payload.event_started) {
+    const startedEventId = String(payload.event_id || "").trim();
+    pushHistoryMessage(
+      startedEventId
+        ? `检测作弊事件开始: ${startedEventId}`
+        : "检测作弊事件开始",
+    );
+  }
+  if (payload.event_finished || (!payload.cheating_detected && lastCheatingDetected)) {
+    pushHistoryMessage("检测作弊事件结束");
+  }
+  if (alertSignature && alertSignature !== lastAlertSignature) {
+    pushHistoryMessage(`告警记录: ${alertSignature}`, { cooldownMs: 4000 });
+  }
+  if (payload.event_persisted) {
+    const persistedEventId = String(payload.event_id || "").trim();
+    pushHistoryMessage(
+      persistedEventId
+        ? `作弊事件记录已保存: ${persistedEventId}`
+        : "作弊事件记录已保存",
+    );
+    if (payload.event_clip_path) {
+      pushHistoryMessage(`事件视频已导出: ${payload.event_clip_path}`);
+    }
+    if (payload.event_raw_frames_path) {
+      pushHistoryMessage(`事件原始帧已导出: ${payload.event_raw_frames_path}`);
+    }
+    loadRecentExports().catch((err) => console.error(err));
+  }
+  lastCheatingDetected = Boolean(payload.cheating_detected);
+  lastAlertSignature = alertSignature;
+
+  if (payload.cheating_detected) {
+    setBadge("检测到可疑行为", "alert");
+  } else if (running) {
+    setBadge("检测进行中", "active");
+  }
+}

+function extractPosePoints(person) {
+  // 兼容格式:
+  // 1) [[x,y,...], ...]
+  // 2) [{x,y,...}, ...]
+  // 3) {keypoints:[...]} / {landmarks:[...]}
+  // 4) {0:[x,y],1:[x,y],...}
+  let raw = person;
+  if (person && typeof person === "object" && !Array.isArray(person)) {
+    if (Array.isArray(person.keypoints)) {
+      raw = person.keypoints;
+    } else if (Array.isArray(person.landmark)) {
+      raw = person.landmark;
+    } else if (Array.isArray(person.landmarks)) {
+      raw = person.landmarks;
+    } else {
+      raw = Object.values(person);
+    }
+  }
+  if (!Array.isArray(raw)) {
+    return [];
+  }
+
+  const points = [];
+  raw.forEach((p) => {
+    if (Array.isArray(p) && p.length >= 2) {
+      const x = Number(p[0]);
+      const y = Number(p[1]);
+      if (Number.isFinite(x) && Number.isFinite(y)) {
+        const score2 = Number(p[2]);
+        const score3 = Number(p[3]);
+        points.push({
+          x,
+          y,
+          visibility: Number.isFinite(score2) ? score2 : (Number.isFinite(score3) ? score3 : null),
+        });
+      }
+      return;
+    }
+    if (p && typeof p === "object") {
+      const x = Number(p.x);
+      const y = Number(p.y);
+      if (Number.isFinite(x) && Number.isFinite(y)) {
+        points.push({
+          x,
+          y,
+          visibility: Number.isFinite(Number(p.visibility)) ? Number(p.visibility) : null,
+          index: Number.isFinite(Number(p.index)) ? Number(p.index) : null,
+        });
+      }
+    }
+  });
+  return points;
+}

+function findPoseAnchor(points, visibleIndexSet) {
+  for (let index = 0; index < points.length; index += 1) {
+    const pt = points[index];
+    if (!visibleIndexSet.has(index)) {
+      continue;
+    }
+    if (isRenderablePoint(pt)) {
+      return pt;
+    }
+  }
+  return null;

+function resolveCoordinateSpace(person, points) {
+  if (person && typeof person === "object" && person.coordinate_space) {
+    return String(person.coordinate_space);
+  }
+  return isNormalizedPose(points) ? "frame_normalized" : "frame_pixels";
+}

+function isNormalizedPose(points) {
+  if (!points.length) {
+    return false;
+  }
+  let inRange = 0;
+  points.forEach((p) => {
+    if (p && Number.isFinite(p.x) && Number.isFinite(p.y) && p.x >= 0 && p.x <= 1.2 && p.y >= 0 && p.y <= 1.2) {
+      inRange += 1;
+    }
+  });
+  return inRange / points.length >= 0.8;
+}
+
+function isRenderablePoint(pt) {
+  if (!pt || !Number.isFinite(pt.x) || !Number.isFinite(pt.y)) {
+    return false;
+  }
+  if (pt.visibility !== null && pt.visibility !== undefined && pt.visibility < 0.35) {
+    return false;
+  }
+  // MediaPipe 无效点常见为 (0,0)
+  if (Math.abs(pt.x) < 1e-6 && Math.abs(pt.y) < 1e-6) {
+    return false;
+  }
+  return true;
+}

+function getContainTransform(canvasW, canvasH, srcW, srcH) {
+  const scale = Math.min(canvasW / Math.max(1, srcW), canvasH / Math.max(1, srcH));
+  const drawW = srcW * scale;
+  const drawH = srcH * scale;
+  const offsetX = (canvasW - drawW) / 2;
+  const offsetY = (canvasH - drawH) / 2;
+  return { scale, offsetX, offsetY };
+}
+
+function toCanvasPoint(x, y, srcW, srcH, transform, coordinateSpace) {
+  const normalized = coordinateSpace === "frame_normalized";
+  const px = normalized ? x * srcW : x;
+  const py = normalized ? y * srcH : y;
+  return [
+    transform.offsetX + px * transform.scale,
+    transform.offsetY + py * transform.scale,
+  ];
+}

+function drawFrameSnapshot() {
+  const ctx = frameCanvasEl.getContext("2d");
+  const w = frameCanvasEl.width;
+  const h = frameCanvasEl.height;
+  ctx.clearRect(0, 0, w, h);
+
+  if (!latestFrameBitmap) {
+    return;
+  }
+
+  const srcW = latestFrameBitmap.width || w;
+  const srcH = latestFrameBitmap.height || h;
+  const transform = getContainTransform(w, h, srcW, srcH);
+  ctx.drawImage(
+    latestFrameBitmap,
+    0,
+    0,
+    srcW,
+    srcH,
+    transform.offsetX,
+    transform.offsetY,
+    srcW * transform.scale,
+    srcH * transform.scale,
+  );
+}

+async function sendFrameForInfer() {
+  if (!running || !videoEl.videoWidth || !videoEl.videoHeight || inferInFlight) {
+    return;
+  }
+  inferInFlight = true;
+  try {
+    const requestId = ++latestRequestId;
+    const temp = document.createElement("canvas");
+    temp.width = videoEl.videoWidth;
+    temp.height = videoEl.videoHeight;
+    temp.getContext("2d").drawImage(videoEl, 0, 0, temp.width, temp.height);
+
+    const blob = await new Promise((resolve) =>
+      temp.toBlob(resolve, "image/jpeg", Number(settings.jpeg_quality || 0.8))
+    );
+    if (!blob) {
+      return;
+    }
+
+    const formData = new FormData();
+    formData.append("file", blob, "frame.jpg");
+
+    const response = await fetch(`/api/webui/infer?threshold=${encodeURIComponent(settings.threshold)}`, {
+      method: "POST",
+      body: formData,
+    });
+    if (!response.ok) {
+      throw new Error(await response.text());
+    }
+    const payload = await response.json();
+    if (requestId !== latestRequestId) {
+      return;
+    }
+
+    latestFrameBitmap = typeof createImageBitmap === "function" ? await createImageBitmap(temp) : temp;
+    updateCameraInfoFromPayload(payload);
+    frameCounter += 1;
+    if (payload.cheating_detected) {
+      detectionCounter += 1;
+    }
+    metricFrames.textContent = String(frameCounter);
+    metricDetections.textContent = String(detectionCounter);
+    metricLatency.textContent = Number(payload.detection_time || 0).toFixed(2);
+
+    const now = performance.now();
+    const dt = Math.max(1, now - lastTick);
+    lastTick = now;
+    metricFps.textContent = (1000 / dt).toFixed(1);
+
+    drawPayload(payload);
+  } finally {
+    inferInFlight = false;
+  }
+}
